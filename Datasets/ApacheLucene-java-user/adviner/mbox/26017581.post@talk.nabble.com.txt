> Use this one:
> 
>  
> 
> String fieldname="BookTitle";
> 
>  
> 
> fieldname = fieldname.intern(); // because of this we need no
> String.equals()
> 
> TermEnum te = IndexReader.terms(new Term(fieldname, ""));
> 
> do {
> 
>       Term term = te.term();
> 
>       if (term == null || term.field() != fieldname) break;
> 
>       System.out.println(term.text());
> 
> } while (te.next());
> 
> te.close();
>
