> >     >
> >     >            <docFactory enabled="true" type="binary">
> >     >
> >     >
<class>net.grcomputing.opencms.search.lucene.BodylessDocument</class>

> >     > For the interested people:

> >     >
> >     >
<class>com.opencms.modules.homepage.news.NewsContentDefinition</class
> >     >>
