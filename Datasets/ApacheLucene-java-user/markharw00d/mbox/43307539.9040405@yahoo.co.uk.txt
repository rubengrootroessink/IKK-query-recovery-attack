Or using XMLEncoder:

        HashMap map=new HashMap();
        map.put("foo","bar");
        ByteArrayOutputStream baos=new ByteArrayOutputStream();
        XMLEncoder encoder =new XMLEncoder(baos);
        encoder.writeObject(map);
        encoder.flush();
        System.out.println(baos.toString());
